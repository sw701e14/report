$ P_i \times \tau_M = P_{i+1}$\\

\begin{itemize}
	\item \textbf{$ P_i $} the initial probability
	\item \textbf{$ \tau_M $} - the markov transition function 
\end{itemize}
		
		

\[
(\underset{h_1}{8},
\underset{d_1}{2},
\underset{h_2}{6},

\end{bmatrix}

(
\underset{h_1}{6.3},
\underset{d_1}{3.6},

\]

\[
